==================== 3. الخوارزميات الأساسية ==================== /** * تحويل PPM إلى كغ/هكتار * @param {number} ppm - قيمة العنصر في التربة (جزء في المليون) * @returns {number} - القيمة بالكيلوغرام لكل هكتار */ function ppmToKgPerHa(ppm) { // معامل التحويل: 1 ppm = 2.4 كغ/هكتار (لعمق تربة 20 سم، كثافة 1.2 طن/م³) return ppm * 2.4; } /** * حساب الكمية المتاحة من العنصر في التربة * @param {number} ppm - قيمة العنصر في التربة * @returns {number} - الكمية المتاحة بالكغ/هكتار */ function availableNutrient(ppm) { return ppmToKgPerHa(ppm); } /** * حساب كمية العنصر المطلوبة لإنتاج محصول معين * @param {object} crop - بيانات المحصول * @param {number} targetYield - المردود المستهدف (طن/هكتار) * @returns {object} - الكميات المطلوبة من كل عنصر (كغ/هكتار) */ function requiredNutrientsForYield(crop, targetYield) { const required = {}; for (const [nutrient, perTon] of Object.entries(crop.nutrientPerTon)) { required[nutrient] = perTon * targetYield; } return required; } /** * حساب العناصر الناقصة حسب قانون الحد الأدنى لليبج * @param {object} available - العناصر المتاحة في التربة (كغ/هكتار) * @param {object} required - العناصر المطلوبة للمحصول (كغ/هكتار) * @returns {object} - العناصر الناقصة والحد الأدنى */ function calculateDeficiencies(available, required) { const deficiencies = {}; let minRatio = Infinity; let limitingElement = null; for (const nutrient of Object.keys(required)) { const availableKg = available[nutrient] || 0; const requiredKg = required[nutrient]; const ratio = availableKg / requiredKg; if (ratio < minRatio) { minRatio = ratio; limitingElement = nutrient; } if (availableKg < requiredKg) { deficiencies[nutrient] = requiredKg - availableKg; } else { deficiencies[nutrient] = 0; } } return { deficiencies, limitingElement, limitingRatio: minRatio }; } /** * حساب خصم المردود بسبب pH و EC * @param {object} crop - بيانات المحصول * @param {number} pH - قيمة pH الحالية * @param {number} EC - قيمة EC الحالية (mS/cm) * @returns {number} - نسبة الخصم (0-1) */ function calculatePenalty(crop, pH, EC) { let penalty = 0; // خصم pH if (pH < crop.idealPH.min) { const deviation = crop.idealPH.min - pH; penalty += deviation * crop.penaltyPH; } else if (pH > crop.idealPH.max) { const deviation = pH - crop.idealPH.max; penalty += deviation * crop.penaltyPH; } // خصم EC if (EC < crop.idealEC.min) { const deviation = crop.idealEC.min - EC; penalty += deviation * crop.penaltyEC; } else if (EC > crop.idealEC.max) { const deviation = EC - crop.idealEC.max; penalty += deviation * crop.penaltyEC; } return Math.min(penalty, 0.8); // الحد الأقصى 80% خصم } /** * حساب المردود المتوقع * @param {object} crop - بيانات المحصول * @param {number} limitingRatio - نسبة العنصر الأكثر نقصاً * @param {number} penalty - نسبة الخصم بسبب pH/EC * @returns {number} - المردود المتوقع (طن/هكتار) */ function calculateExpectedYield(crop, limitingRatio, penalty) { const baseYield = crop.maxYieldTonPerHa; const yieldFactor = Math.min(limitingRatio, 1); const finalFactor = yieldFactor * (1 - penalty); return baseYield * finalFactor; } // ==================== 4. الدالة الرئيسية لحساب المردودية ==================== /** * حساب المردودية الكامل للمحصول بناءً على تحليل التربة * @param {string} cropKey - مفتاح المحصول (مثل "Tomato") * @param {object} soilAnalysis - بيانات تحليل التربة * @param {number} soilAnalysis.N - النيتروجين (ppm) * @param {number} soilAnalysis.P - الفوسفور (ppm) * @param {number} soilAnalysis.K - البوتاسيوم (ppm) * @param {number} soilAnalysis.Ca - الكالسيوم (ppm) - اختياري * @param {number} soilAnalysis.Mg - المغنيسيوم (ppm) - اختياري * @param {number} soilAnalysis.S - الكبريت (ppm) - اختياري * @param {number} soilAnalysis.pH - درجة الحموضة * @param {number} soilAnalysis.EC - الملوحة (mS/cm) * @param {string} language - اللغة ("ar", "en", "fr") * @returns {object} - النتائج الكاملة */ function calculateYield(cropKey, soilAnalysis, language = "ar") { // التحقق من وجود المحصول const crop = CropDatabase[cropKey]; if (!crop) { return { error: `Crop "${cropKey}" not found in database` }; } const lang = language; const t = i18n[lang]; // 1. تحويل PPM إلى كغ/هكتار const availableNutrients = { N: availableNutrient(soilAnalysis.N || 0), P: availableNutrient(soilAnalysis.P || 0), K: availableNutrient(soilAnalysis.K || 0), Ca: availableNutrient(soilAnalysis.Ca || 0), Mg: availableNutrient(soilAnalysis.Mg || 0), S: availableNutrient(soilAnalysis.S || 0) }; // 2. حساب المردود المستهدف (نستخدم المردود الأقصى كهدف أولي) const targetYield = crop.maxYieldTonPerHa; const requiredNutrients = requiredNutrientsForYield(crop, targetYield); // 3. حساب العناصر الناقصة (قانون ليبج) const { deficiencies, limitingElement, limitingRatio } = calculateDeficiencies(availableNutrients, requiredNutrients); // 4. حساب خصم pH و EC const penalty = calculatePenalty(crop, soilAnalysis.pH, soilAnalysis.EC); // 5. حساب المردود المتوقع const expectedYield = calculateExpectedYield(crop, limitingRatio, penalty); // 6. إعادة حساب العناصر الناقصة بناءً على المردود الفعلي const actualRequired = requiredNutrientsForYield(crop, expectedYield); const finalDeficiencies = {}; for (const nutrient of Object.keys(actualRequired)) { const available = availableNutrients[nutrient] || 0; const required = actualRequired[nutrient]; if (available < required) { finalDeficiencies[nutrient] = Math.ceil(required - available); } else { finalDeficiencies[nutrient] = 0; } } // 7. حساب كمية السماد المطلوبة (بالكيلوغرام لكل هكتار) const fertilizerRecommendations = {}; if (finalDeficiencies.N > 0) fertilizerRecommendations.N = finalDeficiencies.N; if (finalDeficiencies.P > 0) fertilizerRecommendations.P = finalDeficiencies.P; if (finalDeficiencies.K > 0) fertilizerRecommendations.K = finalDeficiencies.K; // 8. توصيات لتحسين التربة const recommendations = []; // توصيات pH if (soilAnalysis.pH < crop.idealPH.min) { const limeNeeded = (crop.idealPH.min - soilAnalysis.pH) * 2.5; recommendations.push({ type: "pH", issue: `pH منخفض (${soilAnalysis.pH})`, solution: `أضف ${limeNeeded.toFixed(1)} طن/هكتار من الجير الزراعي`, severity: "high" }); } else if (soilAnalysis.pH > crop.idealPH.max) { const sulfurNeeded = (soilAnalysis.pH - crop.idealPH.max) * 0.5; recommendations.push({ type: "pH", issue: `pH مرتفع (${soilAnalysis.pH})`, solution: `أضف ${sulfurNeeded.toFixed(1)} طن/هكتار من الكبريت الزراعي`, severity: "high" }); } // توصيات EC if (soilAnalysis.EC < crop.idealEC.min) { recommendations.push({ type: "EC", issue: `ملوحة منخفضة (${soilAnalysis.EC} mS/cm)`, solution: "أضف أسمدة عضوية لتحسين النشاط الميكروبي", severity: "medium" }); } else if (soilAnalysis.EC > crop.idealEC.max) { const leachingWater = (soilAnalysis.EC - crop.idealEC.max) * 500; recommendations.push({ type: "EC", issue: `ملوحة مرتفعة (${soilAnalysis.EC} mS/cm)`, solution: `غسل التربة بـ ${leachingWater.toFixed(0)} م³/هكتار من الماء العذب`, severity: "high" }); } // توصيات العناصر الناقصة if (finalDeficiencies.N > 0) { recommendations.push({ type: "N", issue: `نقص نيتروجين (${finalDeficiencies.N} كغ/هكتار)`, solution: `أضف ${Math.ceil(finalDeficiencies.N / 0.46)} كغ/هكتار من اليوريا (46% N)`, severity: finalDeficiencies.N > 50 ? "high" : "medium" }); } if (finalDeficiencies.P > 0) { recommendations.push({ type: "P", issue: `نقص فوسفور (${finalDeficiencies.P} كغ/هكتار)`, solution: `أضف ${Math.ceil(finalDeficiencies.P / 0.18)} كغ/هكتار من سوبر فوسفات (18% P₂O₅)`, severity: finalDeficiencies.P > 30 ? "high" : "medium" }); } if (finalDeficiencies.K > 0) { recommendations.push({ type: "K", issue: `نقص بوتاسيوم (${finalDeficiencies.K} كغ/هكتار)`, solution: `أضف ${Math.ceil(finalDeficiencies.K / 0.5)} كغ/هكتار من كبريتات البوتاسيوم (50% K₂O)`, severity: finalDeficiencies.K > 50 ? "high" : "medium" }); } // 9. إرجاع النتائج return { success: true, crop: { key: cropKey, nameAr: crop.nameAr, nameEn: crop.nameEn, nameFr: crop.nameFr }, soilAnalysis: { nutrients: { N: { ppm: soilAnalysis.N, kgHa: availableNutrients.N }, P: { ppm: soilAnalysis.P, kgHa: availableNutrients.P }, K: { ppm: soilAnalysis.K, kgHa: availableNutrients.K }, Ca: { ppm: soilAnalysis.Ca, kgHa: availableNutrients.Ca }, Mg: { ppm: soilAnalysis.Mg, kgHa: availableNutrients.Mg }, S: { ppm: soilAnalysis.S, kgHa: availableNutrients.S } }, pH: soilAnalysis.pH, EC: soilAnalysis.EC }, yield: { expected: expectedYield.toFixed(1), maxPotential: crop.maxYieldTonPerHa, limitingFactor: limitingElement, limitingRatio: limitingRatio.toFixed(2), penalty: (penalty * 100).toFixed(1) }, deficiencies: finalDeficiencies, fertilizer: fertilizerRecommendations, recommendations: recommendations, i18n: t }; } // ==================== 5. مثال توضيحي للاستخدام ==================== // بيانات تحليل التربة const soilSample = { N: 45, // ppm P: 18, // ppm K: 85, // ppm Ca: 280, // ppm Mg: 45, // ppm S: 12, // ppm pH: 6.2, EC: 1.4 }; console.log("=== Results in English ==="); console.log(resultEn); // حساب المردودية بالفرنسية const resultFr = calculateYield("Tomato", soilSample, "fr"); console.log("=== Résultats en Français ==="); console.log(resultFr); // ==================== 6. دالة لتوليد نص منسق للعرض ==================== function formatResults(result) { if (!result.success) return `Error: ${result.error}`; const t = result.i18n;
